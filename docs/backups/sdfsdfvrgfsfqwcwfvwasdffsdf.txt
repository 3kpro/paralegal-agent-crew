Python:

```python
from google.cloud import aiplatform_v1

client = aiplatform_v1.PredictionServiceClient()
endpoint = (
    "projects/veomachine-475623/locations/us-central1/"
    "publishers/google/models/veo-3.1-generate-preview:predictLongRunning"
)

body = {
    "instances": [{
        "prompt": "5-sec cinematic drone sunset ocean",
        "aspectRatio": "16:9",
        "duration": 5.0,
        "resolution": "1080p"
    }],
    "parameters": {
        "storageUri": "gs://YOUR_BUCKET/videos/",
        "responseCount": 1
    }
}

resp = client.http_request(
    method="POST",
    url=f"https://us-central1-aiplatform.googleapis.com/v1/{endpoint}",
    json=body
)
print(resp)
```
